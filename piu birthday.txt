<script>
    // 1. Password Verification
    function checkPassword() {
        const passwordInput = document.getElementById('password-input').value;
        const errorMsg = document.getElementById('error-msg');
        
        // Set password for Piu's birthday (21072007)
        if(passwordInput === '21072007' || passwordInput.toLowerCase() === 'piu' || passwordInput.toLowerCase() === 'piona') {
            errorMsg.textContent = '';
            nextScreen('screen-question');
        } else {
            errorMsg.textContent = '❌ Wrong code! Try again! (Hint: 21072007 or Piu)';
        }
    }

    // Allow Enter key to submit password
    document.addEventListener('DOMContentLoaded', function() {
        document.getElementById('password-input').addEventListener('keypress', function(e) {
            if(e.key === 'Enter') checkPassword();
        });
    });

    // 2. Navigation Control
    function nextScreen(screenId) {
        document.querySelectorAll('.screen').forEach(screen => screen.classList.remove('active'));
        document.getElementById(screenId).classList.add('active');
    }

    // 3. The Elusive "No" Button Strategy
    let noClickCount = 0;
    function moveNoButton() {
        const noBtn = document.getElementById('no-btn');
        const x = Math.random() * 180 - 90;
        const y = Math.random() * 100 - 50;
        noBtn.style.transform = `translate(${x}px, ${y}px)`;
    }

    function handleNoClick() {
        noClickCount++;
        const gif = document.getElementById('question-gif');
        const heading = document.getElementById('question-text');
        
        if(noClickCount === 1) {
            heading.innerHTML = "How dare you! 😤 Try again, Piu!";
            gif.src = "https://media.giphy.com/media/v1.Y2lkPTc5MGI3NjExOHY5ZzZ2ZHJkZXR2ODd2dms5bnBmcms5cWF6Nms4dnN4NDB2a29xayZlcD12MV9pbnRlcm5hbF9naWZfYnlfaWQmY3Q9cw/kL1SS9LekR08E/giphy.gif";
        } else if(noClickCount === 2) {
            heading.innerHTML = "Accha fine! No more trying 😒";
        } else if(noClickCount >= 3) {
            heading.innerHTML = "Okay fine, you have NO choice! 😉💖";
            document.getElementById('no-btn').style.display = 'none';
            setTimeout(() => nextScreen('screen-letter'), 2000);
        }
    }

    // 4. Letter Interaction
    function openLetter() {
        document.getElementById('envelope').style.display = 'none';
        document.getElementById('secret-letter').style.display = 'block';
    }
</script>
